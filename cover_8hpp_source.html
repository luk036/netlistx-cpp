32template <typename ViolateFunc, typename WeightMap, typename SolutionSet>
33auto pd_cover(ViolateFunc violate, WeightMap& weight, SolutionSet& soln)
34 -> std::pair<SolutionSet, typename WeightMap::mapped_type> {
35 using CostType = typename WeightMap::mapped_type;
36 using NodeType = typename SolutionSet::value_type;
37
38 CostType total_dual_cost = 0;
39 auto gap = weight; // copy weights
40 std::vector<NodeType> added_order;
41
42 // Phase 1: Primal-Dual Selection
43 for (auto&& violate_set : violate()) {
44 if (violate_set.empty()) continue;
45
46 auto min_vtx
47 = *std::min_element(violate_set.begin(), violate_set.end(),
48 [&](const auto& v1, const auto& v2) { return gap[v1] < gap[v2]; });
49 auto min_val = gap[min_vtx];
50
51 if (!soln.contains(min_vtx)) {
52 soln.insert(min_vtx);
53 added_order.emplace_back(min_vtx);
54 }
55
56 total_dual_cost += min_val;
57
58 for (const auto& vtx : violate_set) {
59 gap[vtx] -= min_val;
60 }
61 }
62
63 // Phase 2: Reverse-Delete Post-Processing
64 for (auto it = added_order.rbegin(); it != added_order.rend(); ++it) {
65 soln.erase(*it);
66 bool is_redundant = true;
67 for (auto&& check_set : violate()) {
68 if (!check_set.empty()) {
69 is_redundant = false;
70 break;
71 }
72 }
73 if (!is_redundant) {
74 soln.insert(*it);
75 }
76 }
77
78 CostType final_prml_cost = 0;
79 for (const auto& vtx : soln) {
80 final_prml_cost += weight[vtx];
81 }
82
83 assert(total_dual_cost <= final_prml_cost);
84 return std::make_pair(soln, final_prml_cost);
85}
