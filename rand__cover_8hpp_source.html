78template <typename Hypergraph, typename WeightMap, typename RNG>
79auto rand_hyper_vertex_cover_trial(const Hypergraph& hyprgraph, const WeightMap& weight,
80 const py::set<typename Hypergraph::node_t>& coverset, RNG& rng)
81 -> std::pair<py::set<typename Hypergraph::node_t>, typename WeightMap::mapped_type> {
82 using node_t = typename Hypergraph::node_t;
83 using CostType = typename WeightMap::mapped_type;
84
85 py::set<node_t> soln = coverset.copy();
86 std::vector<node_t> added_order;
87 std::uniform_real_distribution<double> dist(0.0, 1.0);
88
89 for (const auto& net : hyprgraph.nets) {
90 // Copy vertices to a vector for random access
91 std::vector<node_t> vertices(hyprgraph.gr[net].begin(), hyprgraph.gr[net].end());
92
93 // Skip nets already covered
94 bool covered = false;
95 for (const auto& v : vertices) {
96 if (soln.contains(v)) {
97 covered = true;
98 break;
99 }
100 }
101 if (covered || vertices.empty()) {
102 continue;
103 }
104
105 // Generalized Pitt rule: P(pick vi) ~ 1/w(vi)
106 double total_inv{};
107 std::vector<double> inv_weights;
108 inv_weights.reserve(vertices.size());
109 for (const auto& v : vertices) {
110 const double inv = 1.0 / static_cast<double>(weight[v]);
111 inv_weights.push_back(inv);
112 total_inv += inv;
113 }
114
115 const double r = dist(rng);
116 double cumulative{};
117 node_t chosen{};
118 for (std::size_t i = 0; i < vertices.size(); ++i) {
119 cumulative += inv_weights[i] / total_inv;
120 if (r < cumulative) {
121 chosen = vertices[i];
122 break;
123 }
124 }
125
126 soln.insert(chosen);
127 added_order.push_back(chosen);
128 }
129
130 // Phase 2: Reverse-Delete Post-Processing
131 auto is_covered = [&]() -> bool {
132 for (const auto& net : hyprgraph.nets) {
133 bool net_covered = false;
134 for (const auto& v : hyprgraph.gr[net]) {
135 if (soln.contains(v)) {
136 net_covered = true;
137 break;
138 }
139 }
140 if (!net_covered && !hyprgraph.gr[net].empty()) {
141 return false;
142 }
143 }
144 return true;
145 };
146 detail::reverse_delete_cover(soln, added_order, is_covered);
147
148 CostType total_cost{};
149 for (const auto& v : soln) {
150 total_cost += weight[v];
151 }
152 return {std::move(soln), total_cost};
153}
