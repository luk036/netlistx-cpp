63 const std::vector<float>& weights, int /*num_vertices*/,
64 std::vector<unsigned int>& cover, int /*num_words*/,
65 unsigned long long seed) {
66 for (int i = 0; i < num_edges; ++i) {
67 const int u = edges[i * 2];
68 const int v = edges[i * 2 + 1];

73 const unsigned int v_bit = 1u << (v & 31);

76 seed = (seed * 1103515245ULL + 12345ULL) & 0x7FFFFFFFULL;
77 const double rand_val = static_cast<double>(seed) / 2147483648.0;

80 const float w_v = weights[v];
